\subsubsection{\centering  Proof Sketch of the Church-Rosser for $\beta\eta$-Reduction}
In this section, we prove the Church--Rosser Theorem for $\beta\eta$-reduction as done in \cite{selinger2008:lambdanotes}, even though proof is due to Tait and Martin-L\"of. We begin by defining a new relation $M \psred M'$ on terms, called parallel one-step reduction. It is the smallest relation satisfying the following rules:
\begin{mathpar}
\inferrule{P \psred P' \\ N \psred N'}{P N \psred P' N'} \quad {\textsc{ParApp}}


\inferrule{N \psred N'}{\lambda x.N \psred \lambda x.N'} \quad {\textsc{ParAbs}}


\inferrule{Q \psred Q' \\ N \psred N'}{(\lambda x.Q)N \psred Q'[x := N']} \quad \textsc{ParBeta}


\inferrule{P \psred P' \\ x \notin \FV {P}}{\lambda x.(P\,x) \psred P'} \quad \textsc{ParEta}
\end{mathpar}

\noindent The relation $\psred$ \ is also reflexive, $x \psred x$ for every $x$.

\begin{remark}
  In this section, we will not provide explicit proofs of the lemmas. Most are straightforward, follow from induction on terms, and their statements are sensible enough to be understood without detailed derivations.
\end{remark}

\begin{lemma}
  \label{lem:a}
  For all $M, M'$, if $M \betared M'$ then $M \psred M'$.
\end{lemma}

\begin{lemma}
  \label{lem:b}
  For all $M, M'$, if $M \psred M'$ then $M \betareds M'$.
\end{lemma}

\begin{lemma}
  \label{lem:c}
  $\betareds$ is the reflexive, transitive closure of $\psred$.
\end{lemma}

\begin{proof}
  Section 4.4 from \cite{selinger2008:lambdanotes} proves \ref{lem:a}, \ref{lem:b}, \ref{lem:c} in full detail.
\end{proof}

The next goal now is to prove that $\psred$ satisfies the diamond property, since together with lemma \ref{lem:c}, implies the Church--Rosser Property. To this end, we first define the maximal parallel one-step reduct $M^*$ of a term $M$ as follows:
\begin{itemize}
  \item $x^* = x$, for a variable.
  \item $(P\,N)^* = P^*\,N^*$, if $P\,N$ is not a $\beta$-redex.
  \item $((\lambda x.Q)\,N)^* = Q^*[N^*/x]$.
  \item $(\lambda x.N)^* = \lambda x.N^*$, if $\lambda x.N$ is not an $\eta$-redex.
  \item $(\lambda x.(P\,x))^* = P^*$, if $x \notin \FV{P}$.
\end{itemize}
  
\noindent Note that $M^*$ depends only on $M$.

\begin{lemma}
\label{lemma:maximal-parallel}
Whenever $M \psred M'$, then $M' \psred M^*$.
\end{lemma}

\begin{proof}
  Section 4.4 from \cite{selinger2008:lambdanotes}.
\end{proof}


\begin{lemma}
\label{lemma:diamond}
If $M \psred N$ and $M \psred P$, then there exists $Z$ such that $N \psred Z$ and $P \psred Z$.
\end{lemma}

\begin{proof}
Take $Z = M^*$.
\end{proof}

\begin{theorem}
  $\betared$ \ is Church--Rosser.
\end{theorem}

\begin{proof}
  Since $\psred$ satisfies the diamond property (Lemma~\ref{lemma:diamond}),
  its reflexive--transitive closure $\psred^*$ also satisfies the diamond property.
  By Lemma~\ref{lem:c}, $\psred^*$ coincides with $\betareds$.
  Hence $\betareds$ satisfies the diamond property,
  which is exactly the Church--Rosser property for $\betared$.
\end{proof}
